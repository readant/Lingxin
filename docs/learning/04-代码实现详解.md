### 2.1 特征提取器 (feature_extractor.py)

**文件位置**：`src/features/feature_extractor.py`

**功能**：将 MediaPipe 检测的 21 个手部关键点转换为 71 维特征向量。

```python
class FeatureExtractor:
    def extract_features(self, landmarks):
        """
        输入: landmarks - 21个关键点，每个点包含(x, y, z)
        输出: 71维特征向量
        """
        features = []

        for hand_landmarks in landmarks:
            # 步骤1：计算手腕点作为原点
            wrist = hand_landmarks[0]

            # 步骤2：相对化处理 - 消除位置影响
            relative_landmarks = hand_landmarks - wrist

            # 步骤3：计算手指长度
            finger_lengths = []
            fingers = [hand_landmarks[5:9], hand_landmarks[9:13],
                       hand_landmarks[13:17], hand_landmarks[17:21]]
            for finger in fingers:
                length = np.linalg.norm(finger[-1] - finger[0])
                finger_lengths.append(length)

            # 步骤4：计算关节角度
            angles = self.calculate_angles(hand_landmarks)

            # 步骤5：拼接所有特征
            hand_features = np.concatenate([
                relative_landmarks.flatten(),  # 63维
                finger_lengths,                 # 4维
                angles                         # 4维
            ])

            features.append(hand_features)

        # 多只手情况：取平均值
        return np.mean(features, axis=0) if features else np.zeros(71)
```

**特征向量构成**：

| 特征类型 | 维度 | 来源 |
|----------|------|------|
| 相对关键点 | 63 (21×3) | 所有点减去手腕坐标 |
| 手指长度 | 4 | 食指、中指、无名指、小指 |
| 关节角度 | 4 | 4个手指的弯曲角度 |

### 2.2 基础模型类 (base_model.py)

**文件位置**：`src/models/base_model.py`

**功能**：通过模板方法模式统一定义深度学习模型的训练流程。

```python
class BaseModel(ABC, nn.Module):
    @abstractmethod
    def forward(self, x):
        """子类必须实现前向传播"""
        pass

    def train_model(self, train_loader, val_loader=None, epochs=50, lr=0.001):
        """
        通用训练流程（模板方法）
        子类不需要重写此方法
        """
        criterion = self._get_criterion()      # 获取损失函数
        optimizer = self._get_optimizer(lr)    # 获取优化器

        for epoch in range(epochs):
            self.train()                       # 训练模式
            running_loss = 0.0

            for inputs, labels in train_loader:
                optimizer.zero_grad()          # 梯度清零
                outputs = self(inputs)         # 前向传播
                loss = criterion(outputs, labels)
                loss.backward()                # 反向传播
                optimizer.step()               # 更新参数
                running_loss += loss.item() * inputs.size(0)

            epoch_loss = running_loss / len(train_loader.dataset)
            print(f'Epoch {epoch+1}/{epochs}, Loss: {epoch_loss:.4f}')

            # 验证（可选）
            if val_loader:
                val_loss, val_accuracy = self._validate(val_loader, criterion)
                print(f'Val Loss: {val_loss:.4f}, Val Accuracy: {val_accuracy:.4f}')

    def _get_criterion(self):
        """获取损失函数 - 子类可重写"""
        return nn.CrossEntropyLoss()

    def _get_optimizer(self, lr):
        """获取优化器 - 子类可重写"""
        return optim.Adam(self.parameters(), lr=lr)
```

**模板方法模式的优势**：
- 消除 LSTM 和 Transformer 的重复代码约 150 行
- 统一训练流程，便于维护
- 子类只需关注模型结构（forward 方法）

### 2.3 LSTM 模型 (lstm_model.py)

**文件位置**：`src/models/lstm_model.py`

**结构图**：

```
输入 (batch, seq_len, features)
    │
    ↓
┌─────────────────┐
│   LSTM 层        │
│  input_size=features │
│  hidden_size=128    │
│  num_layers=2       │
│  dropout=0.2         │
└─────────────────┘
    │
    ↓
取最后一个时序输出
    │
    ↓
┌─────────────────┐
│   全连接层       │
│   128 → 64      │
│   ReLU 激活     │
└─────────────────┘
    │
    ↓
┌─────────────────┐
│   输出层         │
│   64 → num_classes │
└─────────────────┘
    │
    ↓
输出 (batch, num_classes)
```

**核心代码**：

```python
class LSTMModel(BaseModel):
    def __init__(self, input_shape, num_classes):
        super().__init__()
        self.input_size = input_shape[1]   # 特征维度
        self.hidden_size = 128
        self.num_layers = 2

        self.lstm = nn.LSTM(
            input_size=self.input_size,
            hidden_size=self.hidden_size,
            num_layers=self.num_layers,
            batch_first=True,               # (batch, seq, feature)
            dropout=0.2
        )

        self.fc1 = nn.Linear(self.hidden_size, 64)
        self.fc2 = nn.Linear(64, num_classes)

    def forward(self, x):
        # 初始化隐藏状态
        h0 = torch.zeros(self.num_layers, x.size(0), self.hidden_size).to(x.device)
        c0 = torch.zeros(self.num_layers, x.size(0), self.hidden_size).to(x.device)

        # LSTM 前向传播
        out, _ = self.lstm(x, (h0, c0))

        # 取最后一个时序的输出
        out = out[:, -1, :]

        # 全连接层
        out = self.relu(self.fc1(out))
        out = self.fc2(out)
        return out
```

### 2.5 训练器 (trainer.py)

**文件位置**：`src/training/trainer.py`

**功能**：统管所有模型的训练逻辑。

```python
class Trainer:

    @staticmethod
    def train_classifier(X, y, model_type='svm', test_size=0.2):
        """
        训练传统机器学习模型
        """
        # 数据划分
        X_train, X_test, y_train, y_test = train_test_split(...)

        # 标准化
        scaler = StandardScaler()
        X_train_scaled = scaler.fit_transform(X_train)

        # 选择模型
        models = {
            'svm': SVC(kernel='rbf', C=1.0, gamma='scale'),
            'rf': RandomForestClassifier(n_estimators=100, random_state=42),
            'mlp': MLPClassifier(hidden_layer_sizes=(128, 64), max_iter=1000)
        }

        # 训练和评估
        model.fit(X_train_scaled, y_train)
        y_pred = model.predict(scaler.transform(X_test))
        accuracy = np.mean(y_pred == y_test)

        return {'model': model, 'scaler': scaler}, accuracy

    @staticmethod
    def train_deep_learning(X, y, model_class, test_size=0.2, epochs=50, batch_size=32, lr=0.001):
        """
        训练深度学习模型
        """
        # 数据划分
        X_train, X_test, y_train, y_test = train_test_split(...)

        # 转换为 PyTorch 张量
        X_train = torch.tensor(X_train, dtype=torch.float32)
        y_train = torch.tensor(y_train, dtype=torch.long)

        # 创建 DataLoader
        train_dataset = TensorDataset(X_train, y_train)
        train_loader = DataLoader(train_dataset, batch_size=batch_size, shuffle=True)

        # 创建模型
        model = model_class(input_shape, num_classes)

        # 训练（调用 BaseModel 的 train_model）
        model.train_model(train_loader, val_loader, epochs=epochs, lr=lr)

        # 评估
        model.eval()
        with torch.no_grad():
            y_pred = model(X_test)
            _, y_pred = torch.max(y_pred, 1)
            accuracy = (y_pred == y_test).float().mean().item()

        return model, accuracy
```

## 三、训练入口 (train.py)

**文件位置**：`tools/train.py`

**功能**：提供统一的训练入口，根据模型类型自动选择训练方式。

```python
class TrainRunner:
    # 模型映射配置
    MODEL_MAP = {
        'svm': {'type': 'classifier', 'model_class': None},
        'rf': {'type': 'classifier', 'model_class': None},
        'mlp': {'type': 'classifier', 'model_class': None},
        'lstm': {'type': 'deep_learning', 'model_class': LSTMModel},
        'transformer': {'type': 'deep_learning', 'model_class': TransformerModel}
    }

    # 数据文件映射
    DATA_FILES = {
        'classifier': {'X': 'X.npy', 'y': 'y.npy'},
        'deep_learning': {'X': 'X_sequence.npy', 'y': 'y_sequence.npy'}
    }

    def run(self, data_dir, model_type='svm', save_dir='models'):
        # 1. 获取模型配置
        model_info = self.MODEL_MAP[model_type]
        model_type_category = model_info['type']

        # 2. 加载数据
        data_files = self.DATA_FILES[model_type_category]
        X = np.load(os.path.join(data_dir, data_files['X']))
        y = np.load(os.path.join(data_dir, data_files['y']))

        # 3. 训练模型
        if model_type_category == 'classifier':
            model_data, accuracy = Trainer.train_classifier(X, y, model_type)
            Trainer.save_classifier(model_data, path)
        else:
            model_class = model_info['model_class']
            model, accuracy = Trainer.train_deep_learning(X, y, model_class)
            model.save(path)

        print(f'{model_type} model accuracy: {accuracy:.4f}')
```

## 五、扩展指南

### 5.1 添加新模型步骤

1. 在 `src/models/` 下创建新模型类，继承 `BaseModel`
2. 实现 `__init__` 和 `forward` 方法
3. 在 `train.py` 的 `MODEL_MAP` 中注册
4. 在 `trainer.py` 中添加对应的训练逻辑（如果需要）

### 5.2 示例：添加 GRU 模型

```python
# 1. 创建 src/models/gru_model.py
import torch
import torch.nn as nn
from src.models.base_model import BaseModel

class GRUModel(BaseModel):
    def __init__(self, input_shape, num_classes):
        super().__init__()
        self.gru = nn.GRU(input_shape[1], 128, num_layers=2, batch_first=True)
        self.fc = nn.Linear(128, num_classes)

    def forward(self, x):
        _, h = self.gru(x)
        return self.fc(h[-1])

# 2. 在 train.py 中注册
MODEL_MAP = {
    ...
    'gru': {'type': 'deep_learning', 'model_class': GRUModel}
}
```
